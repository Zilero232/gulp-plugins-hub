    üìã Usage

    import GulpFileExclude from "@zilero/gulp-file-exclude";

    gulp.task('filter', () => {
      return gulp.src('src/**/*')
        .pipe(GulpFileExclude({
          patterns: ['.test.', '.spec.'],
          size: [1000, 1000000] // 1KB to 1MB
        }))
        .pipe(gulp.dest('dist'));
    });

    ‚öôÔ∏è Options

    patterns

    • Type: (string | RegExp)[]

    size

    • Type: [number?, number?]
    • Required: false

    • Type: boolean
    • Default: true

    onExclude

    • Type: (file: Vinyl) => Promise<boolean>
    • Required: false
    • Custom exclusion function

    üìù Example

    gulp.task('advanced-filter', () => {
      return gulp.src('src/**/*')
        .pipe(GulpFileExclude({
          patterns: [
            '.test.',
            /\.spec\./,
            'temp-',
            /-draft\./
          ],
          size: [
            1024,    // min 1KB
            1048576  // max 1MB
          ],
          logExcluded: true,
          onExclude: async (file) => {
            // Custom exclusion logic
            return file.stat?.size > 5000000; // Exclude files > 5MB
          }
        }))
        .pipe(gulp.dest('dist'));
    });
